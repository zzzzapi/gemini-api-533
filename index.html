💻 API调用示例

Python示例

from openai import OpenAI

client = OpenAI(
    base_url="https://zzzzapi.com/v1",
    api_key="sk-xxxx"
)
stream = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[{"role": "user", "content": "Say this is a test"}],
    stream=True,
)
for chunk in stream:
    if chunk.choices[0].delta.content is not None:
        print(chunk.choices[0].delta.content, end="")

JavaScript示例

import { OpenAI } from "openai";

const openai = new OpenAI({
  baseURL: "https://zzzzapi.com/v1",
  apiKey: "sk-xxxx",
});

async function main() {
  const stream = await openai.chat.completions.create({
    model: "claude-3.5-sonnet",
    messages: [{ role: "user", content: "Write a short poem about AI" }],
    stream: true,
  });
  
  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || "");
  }
}

main();

Java示例

import com.theokanning.openai.OpenAiApi;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;

import java.util.Arrays;

public class GeminiExample {
    public static void main(String[] args) {
        String token = "sk-xxxx";
        String baseUrl = "https://zzzzapi.com/v1";
        
        OpenAiService service = new OpenAiService(token, baseUrl);
        
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("gemini-1.5-pro")
            .messages(Arrays.asList(new ChatMessage("user", "Tell me about AI advancements")))
            .build();
            
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
